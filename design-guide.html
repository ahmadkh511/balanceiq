<script>
    let isPaymentMethodCash = false;
    let paidAmountModified = false;
    
    function checkPaymentMethod(value) {
        const container = document.getElementById('paid-amount-container');
        const paidInput = document.getElementById('paid-amount');
        
        // استدعاء API للتحقق من نوع طريقة الدفع
        fetch(`/invoice/get-payment-method/${value}/`)
            .then(res => res.json())
            .then(data => {
                if (data.is_cash) {
                    container.classList.add('show');
                    isPaymentMethodCash = true;
                    // تعبئة المبلغ المدفوع تلقائياً بالإجمالي
                    if (!paidAmountModified) {
                        const grandTotal = document.getElementById('grand-total')?.textContent || 0;
                        paidInput.value = parseFloat(grandTotal).toFixed(2);
                    }
                } else {
                    container.classList.remove('show');
                    isPaymentMethodCash = false;
                    paidInput.value = '0';
                    paidAmountModified = false;
                }
            });
    }
    
    // عند تغيير طريقة الدفع
    document.getElementById('purch_payment_method')?.addEventListener('change', function() {
        checkPaymentMethod(this.value);
    });
    
    // عند تعديل المبلغ المدفوع يدوياً
    document.getElementById('paid-amount')?.addEventListener('input', function() {
        paidAmountModified = true;
        updateRemainingAmount();
    });
</script>

function updateFinancialCalculations() {
    let subtotal = 0;
    document.querySelectorAll('.item-total').forEach(t => subtotal += parseFloat(t.value) || 0);
    const tax = (parseFloat(document.getElementById('tax-percentage')?.value) || 0) * subtotal / 100;
    const add = parseFloat(document.getElementById('invoice-addition')?.value) || 0;
    const disc = parseFloat(document.getElementById('invoice-discount')?.value) || 0;
    const total = subtotal + tax + add - disc;
    
    document.getElementById('subtotal-amount').textContent = subtotal.toFixed(2);
    document.getElementById('grand-total').textContent = total.toFixed(2);
}
12. إرشادات المطورين والمصممين
  • استخدام المتغيرات: ممنوع استخدام ألوان ثابتة، استخدم var(--aqua-1) وما شابه.
  • الاستجابة: استخدم نظام الشبكة من Bootstrap 5 (row و col).
  • الأيقونات: جميع الأيقونات من FontAwesome 6.
  • الخطوط: الخط الأساسي هو 'Tajawal', sans-serif.
  • شركة الشحن: عند اختيار شركة شحن، يجب إظهار حقل رقم الشحنة.
  • طريقة الدفع: عند اختيار "نقدي"، يجب إظهار حقل المبلغ المدفوع.
  • الحقول الإلزامية: أضف class="required-field" إلى الـ <label>.
13. إدارة حقول الباركود الديناميكية (Dynamic Barcode Fields)

⚠️ تحذير مهم: عند إضافة صفوف جديدة (بنود) أو تغيير الكمية في فاتورة تحتوي على 50 حقل باركود، يجب الحفاظ على البيانات المدخلة في حقول الباركود وعدم فقدانها. لا يجوز تحديث الصفحة (Page Refresh) أو إعادة إنشاء العناصر بطريقة تؤدي إلى مسح القيم المدخلة.

المشكلة (The Problem)

تخيل أن لديك فاتورة تحتوي على 50 بنداً، ولكل بند 5 حقول باركود. إذا حاولت إضافة بند جديد باستخدام location.reload() أو إعادة تحميل جزء من الصفحة (AJAX with full DOM replacement)، ستفقد جميع البيانات التي أدخلها المستخدم في حقول الباركود الخمسين.

خطأ شائع: استخدام location.reload() بعد إضافة بند جديد.
الحل (The Solution)

يجب استخدام JavaScript النقي (Vanilla JS) لإضافة الصفوف ديناميكياً دون تحديث الصفحة. يتم إلحاق الصف الجديد بنهاية الجدول، ويتم ربط الأحداث اللازمة له (مثل البحث التلقائي وحساب الإجمالي).

// دالة لإضافة بند جديد مع الحفاظ على البيانات الحالية
function addNewItemRow() {
    // الحصول على الجدول الحالي
    const tableBody = document.querySelector('#items-table tbody');
    if (!tableBody) return;
    
    // حساب رقم البند الجديد
    const rowCount = tableBody.querySelectorAll('tr').length;
    const newRowNumber = rowCount + 1;
    
    // إنشاء الصف الجديد (TR)
    const newRow = document.createElement('tr');
    
    // إضافة محتوى الصف الجديد (يجب أن يتطابق مع هيكل الصفوف الأخرى)
    // من المهم هنا نسخ هيكل الصفوف الحالية باستخدام innerHTML أو إنشائها بالعناصر
    // يجب أن تتضمن الخلايا: الرقم، المادة (مع حقل بحث)، الكمية، السعر، الإجمالي، الملاحظات، الصورة، الإجراءات.
    // **الأهم**: يجب أن تحتوي خلية المادة على حقول الباركود بنفس الهيكل.
    newRow.innerHTML = `
        ${newRowNumber}
        
            

                
                

            

            
            

                ${generateBarcodeFields(5)} <!-- افترض وجود 5 حقول باركود -->
            

        
        
        
        
        
        

        
    `;
    
    // إلحاق الصف الجديد بالجدول
    tableBody.appendChild(newRow);
    
    // إعادة ربط الأحداث للعناصر الجديدة:
    // 1. ربط حدث البحث التلقائي على حقل البحث الجديد
    const newSearchInput = newRow.querySelector('.product-search-input');
    if (newSearchInput) {
        attachAutocomplete(newSearchInput, newRow);
    }
    
    // 2. ربط حدث تغيير الكمية أو السعر لحساب الإجمالي
    const qtyInput = newRow.querySelector('.quantity-input');
    const priceInput = newRow.querySelector('.price-input');
    if (qtyInput && priceInput) {
        qtyInput.addEventListener('input', () => updateItemCalculations(newRow));
        priceInput.addEventListener('input', () => updateItemCalculations(newRow));
        updateItemCalculations(newRow); // حساب أولي
    }
    
    // 3. ربط حدث حذف الصف
    const deleteBtn = newRow.querySelector('.delete-row-btn');
    if (deleteBtn) {
        deleteBtn.addEventListener('click', () => {
            newRow.remove();
            updateFinancialCalculations(); // تحديث المجاميع بعد الحذف
        });
    }
    
    // 4. تحديث المجاميع النهائية بعد الإضافة
    updateFinancialCalculations();
}

// دالة مساعدة لتوليد حقول الباركود (عدد محدد)
function generateBarcodeFields(count) {
    let html = '';
    for (let i = 0; i < count; i++) {
        html += `
            

                
            

        `;
    }
    return `
${html}
`;
}
